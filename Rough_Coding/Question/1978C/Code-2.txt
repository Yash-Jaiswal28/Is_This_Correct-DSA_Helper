for _ in range(int(input())):
    n, k = map(int, input().split())
    if k % 2:
        print("No")
    else:
        a = list(range(1, n + 1))
        for i in range(1, n // 2 + 1):
            now = min(k // 2, n - i + 1 - i)
            a[i-1], a[i + now-1] = a[i + now-1], a[i-1]
            k -= now * 2

        if k!=0:
            print("No")
        else:
            print("Yes")
            print(*a)
